1.DECLARE
  v_today DATE:= SYSDATE;
  v_myname VARCHAR2(20):= 'Laura';
  v_lastname VARCHAR2(20):= 'Tileubek';
BEGIN
  DBMS_OUTPUT.PUT_LINE('Hello World!');
  DBMS_OUTPUT.PUT_LINE('My name is '||v_myname||' '||v_lastname);
  DBMS_OUTPUT.PUT_LINE('Today is '||v_today);
END;

2.DECLARE
v_dep_name VARCHAR2(20);
BEGIN
SELECT DEPARTMENT_NAME INTO v_dep_name FROM DEPARTMENTS WHERE DEPARTMENT_ID=(SELECT DEPARTMENT_ID FROM EMPLOYEES WHERE EMPLOYEE_ID=(
SELECT MAX(EMPLOYEE_ID)FROM EMPLOYEES));
DBMS_OUTPUT.PUT_LINE(v_dep_name);
END;

3.DECLARE
v_salary NUMBER;
BEGIN
SELECT AVG(SALARY)  INTO v_salary FROM EMPLOYEES  WHERE DEPARTMENT_ID IN(SELECT DEPARTMENT_ID FROM DEPARTMENTS WHERE LOCATION_ID IN(
SELECT LOCATION_ID FROM LOCATIONS WHERE COUNTRY_ID IN(SELECT COUNTRY_ID FROM COUNTRIES WHERE COUNTRY_NAME='Canada')));
DBMS_OUTPUT.PUT_LINE(v_salary);
END;

4.DECLARE
v_name VARCHAR2(20);
v_surname VARCHAR2(20);
v_salary NUMBER;
BEGIN
SELECT FIRST_NAME, LAST_NAME,SALARY INTO v_name, v_surname, v_salary FROM EMPLOYEES WHERE HIRE_DATE='JAN 24, 2008';
DBMS_OUTPUT.PUT_LINE('First Name: '||v_name);
DBMS_OUTPUT.PUT_LINE('Last Name: '||v_surname);
DBMS_OUTPUT.PUT_LINE('Salary: '||v_salary);
END;

5.DECLARE
v_dep_name VARCHAR2(20);
BEGIN
SELECT DEPARTMENT_NAME INTO v_dep_name FROM DEPARTMENTS WHERE LOCATION_ID=
(SELECT LOCATION_ID FROM LOCATIONS WHERE CITY='London');
DBMS_OUTPUT.PUT_LINE(v_dep_name);
END;
